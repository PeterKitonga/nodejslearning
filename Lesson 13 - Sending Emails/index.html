Sending Emails

In this section, I learned how to send emails using NodeJS.

Sending Emails using nodemailer

In NodeJS, we send emails using the dependency nodemailer(npm install --save nodemailer). This dependency gives us the utilities for sending emails and can be configured to send using any mail client. 

  1. const ejs = require('ejs');
  2. const path = require('path');
  3. const dotenv = require('dotenv');
  4. const nodemailer = require('nodemailer');
  5. const root_directory = require('../utils/path');
  6.  
  7. dotenv.config(); // loads the .env variables
  8.  
  9. const transporter = nodemailer.createTransport({
  10.   host: process.env.MAIL_HOST,
  11.   port: process.env.MAIL_PORT,
  12.   secure: process.env.MAIL_ENCRYPTION, //ssl
  13.   auth: {
  14.     user: process.env.MAIL_USERNAME,
  15.     pass: process.env.MAIL_PASSWORD
  16.   }
  17. });
  18.  
  19. const send = (receiver, subject, message_options, view) => {
  20.   // send mail with defined transport object
  21.   return new Promise((resolve, reject) => {
  22.     /**
  23.      * Using ejs template for mail views
  24.      * 
  25.      * @link https://stackoverflow.com/questions/41304922/sending-ejs-template-using-nodemailer#answer-41337102
  26.      */
  27.     ejs.renderFile(path.join(root_directory, 'views', 'emails', view), message_options, function (err, data) {
  28.       if (err) {
  29.         console.error(err);
  30.         reject(err);
  31.       } else {
  32.         let mail_options = {
  33.           from: `"${process.env.MAIL_FROM_NAME}" ${process.env.MAIL_FROM_ADDRESS}`, // sender address (who sends)
  34.           to: receiver, // list of receivers (who receives)
  35.           subject: subject, // subject line
  36.           html: data // html body
  37.         };
  38.  
  39.         transporter.sendMail(mail_options, (error, info) => {
  40.           if (error) {
  41.             console.error(`Couldn't send mail ${error}`);
  42.             reject(error);
  43.           } else {
  44.             console.log('Message sent: ' + info.response);
  45.             resolve(info.response);
  46.           }
  47.         });
  48.       }
  49.     });
  50.   });
  51. };
  52.  
  53. module.exports = {
  54.   send
  55. };

Once the above is configured, we can now send emails by importing the mail utility file(utils/mailer.js). In this example I set up a welcome email for new users after signup is complete. 

  1. // controllers/auth.js
  2. const bcrypt = require('bcryptjs');
  3. const User = require('../models/user');
  4. const mailer = require('../utils/mailer');
  5.  
  6. const register = (req, res, next) => {
  7.   const {
  8.     email,
  9.     password,
  10.     confirm_password
  11.   } = req.body;
  12.  
  13.   User.findOne({
  14.     email
  15.   }).then(existing => {
  16.     if (existing) {
  17.       req.flash('error', `User with this email '${email}' already exists, please login or use a different email.`);
  18.       return res.redirect('/signup');
  19.     } else {
  20.       // nesting the promise because we dont want this logic to run when a user is found
  21.       return bcrypt.hash(password, 12).then(hashed_password => {
  22.         const user = new User({
  23.           email,
  24.           password: hashed_password,
  25.           cart: {
  26.             items: []
  27.           }
  28.         });
  29.  
  30.         return user.save();
  31.       }).then(result => {
  32.         res.redirect('/login');
  33.  
  34.         // send mail using the mailer utility file
  35.         return mailer.send(email, 'Welcome to Node.js Learning', {
  36.           message: `Welcome ${email} to the Node.js Learning App.`
  37.         }, 'welcome.ejs');
  38.       }).catch(err => console.log(err));
  39.     }
  40.   }).catch(err => console.log(err));
  41. };
Try it

  1. cd assets/scripts/lesson-13
  2.  
  3. npm install // if you have not installed dependencies
  4.  
  5. cp .env.example .env // copy env variables into variable file and change their values
  6.  
  7. npm run serve // starts application
