Authentication

Adding Authentication

In this section, I learned how to create users with encrypted passwords, adding authentication with encrypted passwords, CSRF tokens and flash messages. This helps improve the experience of authenticating and storing a user.

Encrypting Passwords

It is bad practice to store user passwords as plain text. This is why we encrypt user passwords during registeration which will then be used during authentication in the login. Here we use a library called bcryptjs(npm install --save bcryptjs). This library gives us the tools to hash passwords which cannot be decrypted once hashed during user registration. 

  1. // controllers/auth.js
  2. const bcrypt = require('bcryptjs');
  3. const User = require('../models/user');
  4.  
  5. const register = (req, res, next) => {
  6.   const {
  7.     email,
  8.     password,
  9.     confirm_password
  10.   } = req.body;
  11.  
  12.   User.findOne({
  13.     email
  14.   }).then(existing => {
  15.     if (existing) {
  16.       return res.redirect('/signup');
  17.     } else {
  18.       // nesting the promise because we dont want this logic to run when a user is found
  19.       return bcrypt.hash(password, 12).then(hashed_password => {
  20.         const user = new User({
  21.           email,
  22.           password: hashed_password,
  23.           cart: {
  24.             items: []
  25.           }
  26.         });
  27.  
  28.         return user.save();
  29.       }).then(result => {
  30.         res.redirect('/login');
  31.       });
  32.     }
  33.   }).catch(err => console.log(err));
  34. };

To authenticate the user during login we first need to compare the hash strings between the password used in the login and the password stored in the database. 

  1. // controllers/auth.js
  2. const bcrypt = require('bcryptjs');
  3. const User = require('../models/user');
  4.  
  5. const authenticate = (req, res, next) => {
  6.   const {
  7.     email,
  8.     password
  9.   } = req.body;
  10.  
  11.   User.findOne({
  12.       email
  13.     }).then(user => {
  14.       if (!user) {
  15.         return res.redirect('/login');
  16.       } else {
  17.         // nesting the promise because we dont want this logic to run when a user is not found
  18.         return bcrypt.compare(password, user.password).then(matched => {
  19.           if (matched) {
  20.             // stores session data in the request
  21.             req.session.authenticated = true;
  22.             req.session.user = user;
  23.             return req.session.save((err) => {
  24.               if (err) {
  25.                 console.log(err);
  26.               } else {
  27.                 res.redirect('/');
  28.               }
  29.             });
  30.           } else {
  31.             res.redirect('/login');
  32.           }
  33.         }).catch(err => {
  34.           console.log(err);
  35.           res.redirect('/login');
  36.         });
  37.  
  38.       }
  39.     })
  40.     .catch(err => console.log(err));
  41. };
Protecting routes with a middleware

We can create a middleware that can be passed to routes. Route middlewares are executed from left to right, which means we can pass a middleware to protect a route as the first middleware. Like the example below where we create a middleware for checking if a user is authenticated under middlewares/authenticate.js 

  1. // middlewares/authenticate.js
  2. module.exports = (req, res, next) => {
  3.   // if not authenticated, redirect to login
  4.   if (!req.session.authenticated) {
  5.     return res.redirect('/login');
  6.   }
  7.  
  8.   next();
  9. };
  10.  
  11. // routes/admin.js
  12. const express = require('express');
  13.  
  14. const router = express.Router();
  15.  
  16. const authenticate_middleware = require('../middlewares/authenticate');
  17.  
  18. const admin_controller = require('../controllers/admin');
  19.  
  20. router.get('/products/all', authenticate_middleware, admin_controller.getAllProducts);
  21.  
  22. router.get('/products/add', authenticate_middleware, admin_controller.getAddProduct);
  23.  
  24. router.post('/products/store', authenticate_middleware, admin_controller.postProduct);
  25.  
  26. router.get('/products/edit/:_id', authenticate_middleware, admin_controller.getEditProduct);
  27.  
  28. router.put('/products/update/:_id', authenticate_middleware, admin_controller.updateProduct);
  29.  
  30. router.delete('/products/delete/:_id', authenticate_middleware, admin_controller.deleteProduct);
  31.  
  32. module.exports = {
  33.   admin_routes: router
  34. };
Protecting against CSRF(Cross-Site Request Forgery)

A CSRF attack occurs when an unsuspecting user is directed to a third party site that sends requests to your site to hijack their session and perform requests to your site without them knowing. This can be done via a link sent to your email which directs a user to a site that may look like yours. To protect against such attacks we use CSRF Tokens. For this we use a dependency called csurf(npm install --save csurf). We first configure it in the main app.js file like so: 

  1. const csrf = require('csurf');
  2.  
  3. const csrf_protection = csrf();
  4.  
  5. app.use(csrf_protection);
  6.  
  7. // this middleware passes local variables to all routes
  8. // passed before other route middlewares
  9. app.use((req, res, next) => {
  10.   res.locals.authenticated = req.session.authenticated;
  11.   res.locals.csrf_token = req.csrfToken(); // csrf token
  12.  
  13.   next();
  14. });

Thereafter we can now pass a hidden input field to all forms in our app so as to protect it from CSRF attacks. For example in the form of the views/auth/login.ejs file. 

  1. <form class="login-form" action="/authenticate" method="POST">
  2.   <input type="hidden" name="_csrf" value="<%= csrf_token %>">
  3.   <div class="form-control">
  4.     <label for="login-email">E-mail</label>
  5.     <input type="email" name="email" id="login-email" required>
  6.   </div>
  7.   <div class="form-control">
  8.     <label for="login-password">Password</label>
  9.     <input type="password" name="password" id="login-password" required>
  10.   </div>
  11.   <button class="btn" type="submit">Login</button>
  12. </form>
Flashing messages

Flash messages are used to improve the experience of users during submission of forms and when performing an action. These messages can contain errors or even info messages. For this we use a dependency called connect-flash(npm install --save connect-flash). We first pass it to a middleware in the app.js file just right after the session middleware. 

  1. const express = require('express');
  2. const flash = require('connect-flash');
  3. const session = require('express-session');
  4.  
  5. const app = express();
  6.  
  7. app.use(flash());
  8.  
  9. app.use((req, res, next) => {
  10.   ...
  11.  
  12.   // add it as a local variable to be reused across all views
  13.   res.locals.error_message = req.flash('error');
  14.  
  15.   next();
  16. });

With the above in place we can now flash messages into the session like so: 

  1. // controllers/auth.js
  2. const register = (req, res, next) => {
  3.   const {
  4.     email,
  5.     password,
  6.     confirm_password
  7.   } = req.body;
  8.  
  9.   User.findOne({
  10.     email
  11.   }).then(existing => {
  12.     if (existing) {
  13.       // pass a flash message to the session
  14.       req.flash('error', `User with this email '${email}' already exists, please login or use a different email.`);
  15.       return res.redirect('/signup');
  16.     } else {
  17.       // nesting the promise because we dont want this logic to run when a user is found
  18.       return bcrypt.hash(password, 12).then(hashed_password => {
  19.         const user = new User({
  20.           email,
  21.           password: hashed_password,
  22.           cart: {
  23.             items: []
  24.           }
  25.         });
  26.  
  27.         return user.save();
  28.       }).then(result => {
  29.         res.redirect('/login');
  30.       });
  31.     }
  32.   }).catch(err => console.log(err));
  33. };

In the views/auth/signup.ejs view, we can check for any flashed error message and display it like so: 

  1. <main>
  2.   <% if (error_message.length > 0) { %>
  3.   <div class="alert-message alert-message--error"><%= error_message %></div>
  4.   <% } %>
  5.   <form class="login-form" action="/register" method="POST">
  6.     <input type="hidden" name="_csrf" value="<%= csrf_token %>">
  7.     <div class="form-control">
  8.       <label for="signup-email">E-mail</label>
  9.       <input type="email" name="email" id="signup-email" required>
  10.     </div>
  11.     <div class="form-control">
  12.       <label for="signup-password">Password</label>
  13.       <input type="password" name="password" id="signup-password" required>
  14.     </div>
  15.     <div class="form-control">
  16.       <label for="signup-confirm-password">Confirm Password</label>
  17.       <input type="password" name="confirm_password" id="signup-confirm-password" required>
  18.     </div>
  19.     <button class="btn" type="submit">Sign Up</button>
  20.   </form>
  21. </main>
