Understanding Validation

In this section, I learned how to add validation to user inputs using the package express-validator.

Simple Validation

Here we begin by adding a middleware to the desired route and use the check() method provided by express-validator to set up our validation rules. For example, below a validator has been setup to check if the email during signup is valid. 

  1. const express = require('express');
  2. const {
  3.   check
  4. } = require('express-validator');
  5.  
  6. const router = express.Router();
  7. const auth_controller = require('../controllers/auth');
  8.  
  9. router.post(
  10.   '/register',
  11.   check('email')
  12.   .isEmail()
  13.   .withMessage('Please enter a valid email.'),
  14.   auth_controller.register
  15. );

Thereafter, we use the validationResult() method provided by express-validator to check for errors which will be in array form. 

  1. const crypto = require('crypto');
  2. const moment = require('moment');
  3. const bcrypt = require('bcryptjs');
  4. const User = require('../models/user');
  5. const mailer = require('../utils/mailer');
  6. const {
  7.   validationResult
  8. } = require('express-validator');
  9.  
  10. const register = (req, res, next) => {
  11.   const {
  12.     email,
  13.     password,
  14.     confirm_password
  15.   } = req.body;
  16.   const errors = validationResult(req);
  17.  
  18.   // check for validation errors
  19.   if (!errors.isEmpty()) {
  20.     // render the view with the validation error
  21.     return res.status(422).render('auth/signup', {
  22.       page_title: 'Shop | Sign Up',
  23.       route_name: 'auth.signup',
  24.       error_message: errors.array()[0].msg
  25.     });
  26.   }
  27.  
  28.   User.findOne({
  29.     email
  30.   }).then(existing => {
  31.     if (existing) {
  32.       req.flash('error', `User with this email '${email}' already exists, please login or use a different email.`);
  33.       return res.redirect('/signup');
  34.     } else {
  35.       // nesting the promise because we dont want this logic to run when a user is found
  36.       return bcrypt.hash(password, 12).then(hashed_password => {
  37.         const user = new User({
  38.           email,
  39.           password: hashed_password,
  40.           cart: {
  41.             items: []
  42.           }
  43.         });
  44.  
  45.         return user.save();
  46.       }).then(result => {
  47.         res.redirect('/login');
  48.  
  49.         // send mail using the mailer utility file
  50.         return mailer.send(email, 'Welcome to Node.js Learning', {
  51.           message: `Welcome ${email} to the Node.js Learning App.`
  52.         }, 'welcome.ejs');
  53.       }).catch(err => console.log(err));
  54.     }
  55.   }).catch(err => console.log(err));
  56. };
Custom Validation

For a full list of built in validation rules check out validator.js. We can also define custom validators by chaining the custom() method which expects the value as the first parameter and object containing the req, location and path fields 

  1. const express = require('express');
  2. const {
  3.   check
  4. } = require('express-validator');
  5.  
  6. const router = express.Router();
  7. const auth_controller = require('../controllers/auth');
  8.  
  9. router.post(
  10.   '/register',
  11.   check('email')
  12.   .isEmail()
  13.   .withMessage('Please enter a valid email.'),
  14.   check('password')
  15.   .exists()
  16.   .isLength({
  17.     min: 6,
  18.     max: 255
  19.   })
  20.   .withMessage('The password field should be between 6 and 255 characters.'),
  21.   check('confirm_password')
  22.   .custom((value, {
  23.     req
  24.   }) => {
  25.     if (value !== req.body.password) {
  26.       throw new Error('The confirm_password field must have the same value as the password field.');
  27.     }
  28.  
  29.     return true;
  30.   }),
  31.   auth_controller.register
  32. );
Asynchronous Validation

We can validate asynchronously by returning a promise inside a custom validator. This let's express-validator know that it needs to wait and check if the value being validated satisfies the constraints put in place for validation. For example, below we want to check if a user with the email passed during registration exists. The promise will be rejected if the condition is true and a validation error is thrown. 

  1. // routes/auth.js
  2. const express = require('express');
  3. const {
  4.   check,
  5.   body
  6. } = require('express-validator');
  7.  
  8. const router = express.Router();
  9.  
  10. const User = require('../models/user');
  11. const auth_controller = require('../controllers/auth');
  12.  
  13. router.post(
  14.   '/register', [
  15.     check('email')
  16.     .isEmail()
  17.     .withMessage('The email field is required.')
  18.     .custom((value, {
  19.       req
  20.     }) => {
  21.       // here we let express validator know that it need to check asynchronously if the email exists then serve up the respective result
  22.       return User.findOne({
  23.         email: value
  24.       }).then(existing => {
  25.         if (existing) {
  26.           return Promise.reject(
  27.             `User with this email '${value}' already exists, please login or use a different email.`
  28.           );
  29.         }
  30.       });
  31.     }),
  32.     body('password')
  33.     .exists()
  34.     .isLength({
  35.       min: 6,
  36.       max: 255
  37.     })
  38.     .withMessage('The password field should be between 6 and 255 characters.'),
  39.     body('confirm_password')
  40.     .custom((value, {
  41.       req
  42.     }) => {
  43.       if (value !== req.body.password) {
  44.         throw new Error('The confirm_password field must have the same value as the password field.');
  45.       }
  46.  
  47.       return true;
  48.     })
  49.   ],
  50.   auth_controller.register
  51. );
