File Upload & Download

In this section, I learn about handling file uploads and downloads in Node.js. We begin by installing a dependency called multer(npm install --save multer). This dependency allows us to handle multipart/form-data enctypes of forms to allow uploading of files.

Configuring Multer

We begin by setting up options for the multer middleware where we configure how files are stored(storage) and when to accept/reject storing a file(fileFilter). 

  1. const multer = require('multer');
  2. const express = require('express');
  3.  
  4. const app = express();
  5.  
  6. const file_storage = multer.diskStorage({
  7.   destination(req, file, _callback) {
  8.     _callback(null, 'public/storage/files');
  9.   },
  10.   filename(req, file, _callback) {
  11.     _callback(null, `${Date.now()}-${file.originalname}`);
  12.   },
  13. });
  14.  
  15. const fileFilter = (req, file, _callback) => {
  16.   const types = ['image/png', 'image/jpg', 'image/jpeg'];
  17.  
  18.   if (types.includes(file.mimetype)) {
  19.     _callback(null, true);
  20.   } else {
  21.     _callback(null, false);
  22.   }
  23. };
  24.  
  25. // multer middleware which allows multipart/form-data form inputs
  26. app.use(multer({
  27.   storage: file_storage,
  28.   fileFilter
  29. }).single('image'));
Storing files and saving to the database

To be able to store files we first need to add a file picker to the form like below. 

  1. <form class="product-form" action="/admin/products/store" method="POST" novalidate enctype="multipart/form-data">
  2.   <input type="hidden" name="_csrf" value="<%= csrf_token %>">
  3.   <!-- other form elements -->
  4.   <div class="form-control">
  5.     <label for="product-image-url">Image</label>
  6.     <input 
  7.       type="file" 
  8.       name="image"
  9.       id="product-image-url">
  10.   </div>
  11.   <!-- other form elements -->
  12.   <button class="btn" type="submit">Add Product</button>
  13. </form>

With the file picker in place, we can now fetch the file during the POST where we can get details of the file path and name. Here we can construct the accessible path to the file and store it in the database. This way we will be able to view the file from the browser. 

  1. // controllers/admin.js
  2. const Product = require('../models/product');
  3. const {
  4.   validationResult
  5. } = require('express-validator');
  6. const {
  7.   errorHandler
  8. } = require('../utils/error-handler');
  9.  
  10. const postProduct = (req, res, next) => {
  11.   const {
  12.     title,
  13.     price,
  14.     description
  15.   } = req.body;
  16.   const errors = validationResult(req);
  17.   const image = req.file;
  18.  
  19.   if (!errors.isEmpty()) {
  20.     return res.status(422).render('admin/products/add', {
  21.       page_title: 'Shop | Add Product',
  22.       route_name: 'products.add',
  23.       error_message: errors.array(),
  24.       old_input: {
  25.         title,
  26.         price,
  27.         description
  28.       }
  29.     });
  30.   }
  31.  
  32.   if (!image) {
  33.     return res.status(422).render('admin/products/add', {
  34.       page_title: 'Shop | Add Product',
  35.       route_name: 'products.add',
  36.       error_message: 'The image field is required.',
  37.       old_input: {
  38.         title,
  39.         price,
  40.         description
  41.       }
  42.     });
  43.   }
  44.  
  45.   // ["", "/storage/files/1621249822068-fakeimg.png"][1]
  46.   const storage_path = image.path.split('public')[1];
  47.  
  48.   // concatenate app url and storage path
  49.   const image_url = process.env.APP_BASE_URL + storage_path;
  50.  
  51.   const product = new Product({
  52.     title,
  53.     price,
  54.     image_url,
  55.     description,
  56.     user: req.user._id
  57.   });
  58.  
  59.   product.save().then(result => {
  60.     res.redirect('/admin/products/all');
  61.   }).catch(err => errorHandler(err, next));
  62. };
Downloading files

To download files we first need to load and stream the file to the browser. For that we use the inbuilt file system dependency for Node.js. We stream the file because this will improve performance and allows us to pipe the streamed file into the response. We set headers as well to let the browser know what file name to use during download. 

  1. const fs = require('fs');
  2. const path = require('path');
  3. const Order = require('../models/order');
  4. const {
  5.   errorHandler
  6. } = require('../utils/error-handler');
  7.  
  8. const getOrderInvoice = (req, res, next) => {
  9.   const {
  10.     order_id
  11.   } = req.params;
  12.  
  13.   Order.findById(order_id).then(order => {
  14.     if (!order) {
  15.       return errorHandler(new Error('No order found.'), next);
  16.     }
  17.  
  18.     if (order.user.toString() !== req.user._id.toString()) {
  19.       return errorHandler(new Error('Unauthorized download of requested invoice.'), next);
  20.     }
  21.  
  22.     const invoice_name = `invoice-${order_id}.pdf`;
  23.     const invoice_path = path.join('data', 'invoices', invoice_name);
  24.  
  25.     /**
  26.      * Below commented code is not recommended for large files because this 
  27.      * will read the whole file and load it on memory which
  28.      * would impact perfomance.
  29.      */
  30.     // fs.readFile(invoice_path, (err, data) => {
  31.     //     if(err) {
  32.     //         return errorHandler(err, next);
  33.     //     } else {
  34.     //         res.setHeader('Content-Type', 'application/pdf');
  35.     //         res.setHeader('Content-Disposition', `attachment; filename="${invoice_name}"`);
  36.     //         res.send(data);
  37.     //     }
  38.     // });
  39.  
  40.     /**
  41.      * Streaming the file data is the optimal way for downloading file data
  42.      */
  43.     const file = fs.createReadStream(invoice_path);
  44.     res.setHeader('Content-Type', 'application/pdf');
  45.     res.setHeader('Content-Disposition', `attachment; filename="${invoice_name}"`);
  46.  
  47.     /**
  48.      * This will pipe the streamed file data to the response
  49.      */
  50.     file.pipe(res);
  51.   }).catch(err => errorHandler(err, next));
  52. };
Creating files using pdfkit

Pdfkit is a dependency for node.js used for generating pdfs. We first need to install it with the command npm install pdfkit --save. Once installed we use it together with the inbuilt node.js file system library to write a file and add text to the file before downloading it. 

  1. // controllers/shop.js
  2. const fs = require('fs');
  3. const path = require('path');
  4. const PDFDocument = require('pdfkit');
  5. const Order = require('../models/order');
  6. const {
  7.   errorHandler
  8. } = require('../utils/error-handler');
  9.  
  10. const getOrderInvoice = (req, res, next) => {
  11.   const {
  12.     order_id
  13.   } = req.params;
  14.  
  15.   Order.findById(order_id).then(order => {
  16.     if (!order) {
  17.       return errorHandler(new Error('No order found.'), next);
  18.     }
  19.  
  20.     if (order.user.toString() !== req.user._id.toString()) {
  21.       return errorHandler(new Error('Unauthorized download of requested invoice.'), next);
  22.     }
  23.  
  24.     const invoice_document = new PDFDocument();
  25.     const invoice_name = `invoice-${order_id}.pdf`;
  26.     const invoice_path = path.join('data', 'invoices', invoice_name);
  27.  
  28.     res.setHeader('Content-Type', 'application/pdf');
  29.     res.setHeader('Content-Disposition', `attachment; filename="${invoice_name}"`);
  30.  
  31.     invoice_document.pipe(fs.createWriteStream(invoice_path));
  32.     invoice_document.pipe(res);
  33.  
  34.     invoice_document.fontSize(20).text('Order Invoice', {
  35.       underline: true
  36.     });
  37.  
  38.     let total_price = 0;
  39.     order.products.forEach(item => {
  40.       total_price += item.quantity * item.product.price;
  41.       invoice_document.fontSize(14).text(`${item.product.title} ===============> ${item.quantity} x $${item.product.price}`);
  42.     });
  43.  
  44.     invoice_document.fontSize(14).text('--------------------------------------------');
  45.  
  46.     invoice_document.fontSize(18).text(`Total Price ===============> $${total_price}`);
  47.  
  48.     invoice_document.end();
  49.   }).catch(err => errorHandler(err, next));
  50. };
