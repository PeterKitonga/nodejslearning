Working with NoSQL & using MongoDB

In this section, I learned how to configure and connect to a MongoDB database. In the setup below, I am using MongoDB Atlas which is a cloud server for MongoDB databases. It is free and the setup is straight foward. Note: I will not cover the setup of the server here.

Connecting to the database

Before connecting to a MongoDB database we first need to install a MongoDB client called mongodb(npm install mongodb). This should provide us with utilies for connecting to the MongoDB database. In the utils/database.js file, we import the mongo client and pass the connection URI as well as options for the connection. I passed a database variable to hold data for the database which I used in the getDatabase function which we will use in all other files. 

  1. const mongodb = require('mongodb');
  2. const dotenv = require('dotenv');
  3.  
  4. const {
  5.   MongoClient
  6. } = mongodb;
  7. dotenv.config();
  8.  
  9. let database;
  10.  
  11. const mongoConnect = () => {
  12.   return new Promise((resolve, reject) => {
  13.     const client = new MongoClient(
  14.       `mongodb+srv://${process.env.MONGO_USERNAME}:${process.env.MONGO_PASSWORD}@nodejs.j6dzr.mongodb.net/${process.env.MONGO_DATABASE}?retryWrites=true&w=majority`, {
  15.         useNewUrlParser: true,
  16.         useUnifiedTopology: true
  17.       }
  18.     );
  19.  
  20.     client.connect().then(result => {
  21.       console.log('MONGO CONNECTED!');
  22.       database = result.db();
  23.       resolve();
  24.     }).catch(err => reject(err));
  25.   });
  26. };
  27.  
  28. const getDatabase = () => {
  29.   if (database) {
  30.     return database;
  31.   } else {
  32.     throw 'NO DATABASE FOUND!';
  33.   }
  34. };
  35.  
  36. module.exports = {
  37.   mongoConnect,
  38.   getDatabase
  39. };

In the app.js file, I imported the mongoConnect function which I used to run the app once a connection is successfully established with our MongoDB database. 

  1. const {
  2.   mongoConnect
  3. } = require('./utils/database');
  4.  
  5. mongoConnect().then(() => {
  6.   // mount the nodejs app to a port
  7.   app.listen(port, () => {
  8.     console.log(`Server running at: http://127.0.0.1:${port}`);
  9.     console.log('Hit CTRL-C to stop the server');
  10.   });
  11. }).catch(err => console.log(err));
Defining a Model

Once connection is established to MongoDB we can define our model by importing the getDatabase method in the database util file(utils/database.js). Here, we define a class for the model and pass the fields for the model to the constructor. The model will be used to access a collection in the MongoDB database and perform CRUD actions like save, update, delete. 

  1. const {
  2.   getDatabase
  3. } = require('../utils/database');
  4.  
  5. class Product {
  6.   // fields for the collection
  7.   constructor({
  8.     title,
  9.     price,
  10.     image_url,
  11.     description
  12.   }) {
  13.     this.title = title;
  14.     this.price = price;
  15.     this.image_url = image_url;
  16.     this.description = description;
  17.   }
  18. }
  19.  
  20. module.exports = Product;
Inserting data into database

We can now use the Product model to store data to our products collection using the insertOne method. This inserts one document into the collection. In this example, I created a save() method to pass the object passed to the constructor to the collection. 

  1. const {
  2.   getDatabase
  3. } = require('../utils/database');
  4.  
  5. class Product {
  6.   // fields for the collection
  7.   constructor({
  8.     title,
  9.     price,
  10.     image_url,
  11.     description
  12.   }) {
  13.     this.title = title;
  14.     this.price = price;
  15.     this.image_url = image_url;
  16.     this.description = description;
  17.   }
  18.  
  19.   save() {
  20.     return new Promise((resolve, reject) => {
  21.       // gets the database from the  mongo connection
  22.       const database = getDatabase();
  23.  
  24.       // inserts one document to the products collection
  25.       database.collection('products').insertOne(this).then(result => {
  26.         resolve(result);
  27.       }).catch(err => reject(err));
  28.     });
  29.   }
  30. }
  31.  
  32. module.exports = Product;

In the controller, we can now pass the object to be saved to the products collection. 

  1. const postProduct = (req, res, next) => {

Fetching data from database

To fetch documents from the products collection, we first need to define a static method in the model that will access the collection and fetch all the documents. Consider the example below. 

  1. // models/product.js
  2. const {
  3.   getDatabase
  4. } = require('../utils/database');
  5.  
  6. class Product {
  7.   // fields for the collection
  8.   constructor({
  9.     title,
  10.     price,
  11.     image_url,
  12.     description
  13.   }) {
  14.     this.title = title;
  15.     this.price = price;
  16.     this.image_url = image_url;
  17.     this.description = description;
  18.   }
  19.  
  20.   // other class prototypes....
  21.  
  22.   static fetchAll() {
  23.     return new Promise((resolve, reject) => {
  24.       // gets the database from the mongo connection
  25.       const database = getDatabase();
  26.  
  27.       /**
  28.        * 'find()' is a CRUD operation for fetching all(like below) or single documents in the collection
  29.        * 'toArray()' will convert the json object to an array
  30.        * 
  31.        * @link https://docs.mongodb.com/manual/tutorial/query-documents/
  32.        */
  33.       database.collection('products').find({}).toArray().then(result => {
  34.         resolve(result);
  35.       }).catch(err => reject(err));
  36.     });
  37.   }
  38. }
  39.  
  40. module.exports = Product;
  41.  
  42. // controllers/shop.js
  43. const Product = require('../models/product');
  44.  
  45. const getHome = (req, res, next) => {
  46.   Product.fetchAll().then(products => {
  47.     res.render('shop/index', {
  48.       prods: products,
  49.       page_title: 'Shop | Home',
  50.       route_name: 'home'
  51.     });
  52.   }).catch(err => console.log(err));
  53. };

To find a specific document, we use the .find() method where we pass an object that contains the constraints of the query. In the example below use the id(_id) to fetch a single document. 

  1. // models/product.js
  2. const {
  3.   ObjectId
  4. } = require('mongodb'); // import mongo for special functions
  5. const {
  6.   getDatabase
  7. } = require('../utils/database');
  8.  
  9. class Product {
  10.   // fields for the collection
  11.   constructor({
  12.     title,
  13.     price,
  14.     image_url,
  15.     description
  16.   }) {
  17.     this.title = title;
  18.     this.price = price;
  19.     this.image_url = image_url;
  20.     this.description = description;
  21.   }
  22.  
  23.   // other class prototypes....
  24.  
  25.   static findById(product_id) {
  26.     return new Promise((resolve, reject) => {
  27.       // gets the database from the mongo connection
  28.       const database = getDatabase();
  29.  
  30.       /**
  31.        * We use 'next()' to get the last object in the collection which is technically the document we are fetching
  32.        * In MongoDB ids are stored as 'object ids'. We pass the id received in the ObjectId constructor to convert it to the special MongoDB object id
  33.        * 
  34.        * @link https://docs.mongodb.com/manual/tutorial/query-documents/
  35.        */
  36.       database.collection('products').find({
  37.         _id: new ObjectId(product_id)
  38.       }).next().then(result => {
  39.         resolve(result);
  40.       }).catch(err => reject(err));
  41.     });
  42.   }
  43. }
  44.  
  45. module.exports = Product;
  46.  
  47. // controllers/shop.js
  48. const Product = require('../models/product');
  49.  
  50. const getProduct = (req, res, next) => {
  51.   const {
  52.     _id
  53.   } = req.params;
  54.  
  55.   Product.findById(_id).then(product => {
  56.       res.render('shop/view', {
  57.         product,
  58.         page_title: 'Shop | Listing | View',
  59.         route_name: 'list.view'
  60.       });
  61.     })
  62.     .catch(err => console.log(err));
  63. };
Updating data

To update data we use the updateOne() method which accepts two arguments as objects. The first argument contains the constraints for the update query such as updating documents with a specific id, name, description e.t.c. The second argument contains the fiels which are set using the $set field. 

  1. // models/product.js
  2. const {
  3.   ObjectId
  4. } = require('mongodb'); // import mongo for special functions
  5. const {
  6.   getDatabase
  7. } = require('../utils/database');
  8.  
  9. class Product {
  10.   // fields for the collection
  11.   constructor({
  12.     _id,
  13.     title,
  14.     price,
  15.     image_url,
  16.     description
  17.   }) {
  18.     this._id = new ObjectId(_id);
  19.     this.title = title;
  20.     this.price = price;
  21.     this.image_url = image_url;
  22.     this.description = description;
  23.   }
  24.  
  25.   // other class prototypes....
  26.  
  27.   update() {
  28.     return new Promise((resolve, reject) => {
  29.       // gets the database from the mongo connection
  30.       const database = getDatabase();
  31.  
  32.       /**
  33.        * The 'updateOne()' method is used in MongoDB to update one record that satisfies the contraints passed in the first object argument
  34.        * The second object argument are the fields to be updated where we use $set. $currentDate updates the last modified date field to the current date
  35.        * 
  36.        * @link https://docs.mongodb.com/manual/tutorial/update-documents/
  37.        */
  38.       database.collection('products').updateOne({
  39.         _id: this._id
  40.       }, {
  41.         $set: this,
  42.         $currentDate: {
  43.           lastModified: true
  44.         }
  45.       }).then(result => {
  46.         resolve(result);
  47.       }).catch(err => reject(err));
  48.     });
  49.   }
  50. }
  51.  
  52. module.exports = Product;
  53.  
  54. // controllers/admin.js
  55. const Product = require('../models/product');
  56.  
  57. const updateProduct = (req, res, next) => {
  58.   const {
  59.     _id
  60.   } = req.params;
  61.   const updated = req.body;
  62.  
  63.   const product = new Product({
  64.     _id,
  65.     ...updated
  66.   });
  67.  
  68.   product.update().then(result => {
  69.       res.redirect('/admin/products/all');
  70.     })
  71.     .catch(err => console.log(err));
  72. };
Deleting data

Here we use the deleteOne() method to delete one document with the specific constraints(by _id). 

  1. // models/product.js
  2. const {
  3.   ObjectId
  4. } = require('mongodb');
  5. const {
  6.   getDatabase
  7. } = require('../utils/database');
  8.  
  9. class Product {
  10.   // fields for the collection
  11.   constructor({
  12.     _id,
  13.     title,
  14.     price,
  15.     image_url,
  16.     description
  17.   }) {
  18.     this._id = _id ? new ObjectId(_id) : null;
  19.     this.title = title;
  20.     this.price = price;
  21.     this.image_url = image_url;
  22.     this.description = description;
  23.   }
  24.  
  25.   // other prototypes...
  26.  
  27.   static deleteById(product_id) {
  28.     return new Promise((resolve, reject) => {
  29.       // gets the database from the mongo connection
  30.       const database = getDatabase();
  31.  
  32.       /**
  33.        * 'deleteOne' expects one argument as an object with constraints for the query
  34.        * 
  35.        * @link https://docs.mongodb.com/manual/tutorial/remove-documents/
  36.        */
  37.       database.collection('products').deleteOne({
  38.         _id: new ObjectId(product_id)
  39.       }).then(result => {
  40.         resolve(result);
  41.       }).catch(err => reject(err));
  42.     });
  43.   }
  44. }
  45.  
  46. module.exports = Product;
  47.  
  48. // controllers/admin.js
  49. const Product = require('../models/product');
  50.  
  51. const deleteProduct = (req, res, next) => {
  52.   const {
  53.     _id
  54.   } = req.params;
  55.  
  56.   Product.deleteById(_id).then(result => {
  57.       res.redirect('/admin/products/all');
  58.     })
  59.     .catch(err => console.log(err));
  60. };
