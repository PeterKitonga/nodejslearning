In this section, I learned how to create password resets functionality and applied knowledge from the previous(Sending Emails) section.

Password Resets

Here we begin by setting up two fields(reset_token, reset_token_expiration) in the User model which will add them to the users collection. We then add a route middleware to send a reset link via email with the stored token like below: 

  1. const crypto = require('crypto');
  2. const moment = require('moment');
  3. const mailer = require('../utils/mailer');
  4.  
  5. const emailResetLink = (req, res, next) => {
  6.   const {
  7.     email
  8.   } = req.body;
  9.  
  10.   crypto.randomBytes(64, (err, buffer) => {
  11.     if (err) {
  12.       req.flash('error', 'Something went wrong. Please try again.');
  13.  
  14.       return res.redirect('/forgot');
  15.     } else {
  16.       const token = buffer.toString('hex');
  17.  
  18.       User.findOne({
  19.         email
  20.       }).then(existing => {
  21.         if (!existing) {
  22.           req.flash('error', `User with this email '${email}' doesn't exist, please use a different email.`);
  23.  
  24.           return res.redirect('/forgot');
  25.         } else {
  26.           const expiration_date = moment().add(1, 'days').utcOffset(3).format("YYYY-MM-DD HH:mm:ss");
  27.  
  28.           existing.reset_token = token;
  29.           existing.reset_token_expiration = expiration_date;
  30.  
  31.           return existing.save();
  32.         }
  33.       }).then(result => {
  34.         res.redirect('/');
  35.  
  36.         // send mail using the mailer utility file
  37.         return mailer.send(email, 'Reset Password Link for Node.js Learning', {
  38.           message: `We have received a reset password request for ${email} for the Node.js Learning App.`,
  39.           token
  40.         }, 'reset.ejs');
  41.       }).catch(err => console.log(err));
  42.     }
  43.   })
  44. };

The above will send a reset link via email which contains a reset token which will be passed during submission of the new password.
