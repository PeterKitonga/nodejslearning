Modern JavaScript & Node.js

In this section, I learned about ES modules and how to implement them in node.js. ES modules uses the Import/Export syntax.

Setting up ES Modules

In Node.js, using the require/module.exports syntax is the default way of importing and exporting modules. To enable ES modules, we first need to add a configuration in the package.json file. We do this by adding the line as illustrated below. 

  1. {
  2.   "name": "es-modules",
  3.   "version": "1.0.0",
  4.   "description": "Example setup with ES modules",
  5.   "main": "src/app.js",
  6.   "type": "module", // enables ES modules
  7.   "scripts": {
  8.     "serve": "nodemon src/app.js",
  9.     "test": "echo \"Error: no test specified\" && exit 1"
  10.   },
  11.   "author": "Peter Kitonga",
  12.   "license": "ISC",
  13.   "dependencies": {
  14.     "compression": "^1.7.4",
  15.     "dotenv": "^10.0.0",
  16.     "dotenv-expand": "^5.1.0",
  17.     "express": "^4.17.1",
  18.     "helmet": "^4.6.0"
  19.   },
  20.   "devDependencies": {
  21.     "nodemon": "^2.0.12"
  22.   }
  23. }
Using ES Modules

We can now use ES modules by first converting all code to use the import/export syntax like below. 

  1. // src/app.js
  2. import dotenv from "dotenv";
  3. import express from "express";
  4. import {
  5.   fileURLToPath
  6. } from "url";
  7. import {
  8.   join,
  9.   dirname
  10. } from "path";
  11. import dotenvExpand from "dotenv-expand";
  12.  
  13. const __dirname = dirname(fileURLToPath(
  14.   import.meta.url));
  15. dotenvExpand(dotenv.config({
  16.   path: join(__dirname, "../.env")
  17. }));
  18.  
  19. const app = express();
  20. const port = process.env.APP_PORT;
  21.  
  22. import resHandler from "./response-handler.js";
  23.  
  24. app.get("/", resHandler);
  25.  
  26. app.listen(port, () => {
  27.   console.log(`Server running at: ${process.env.APP_BASE_URL}`);
  28.   console.log("Hit CTRL-C to stop the server");
  29. });

This means that our response-handler file would look something like: 

  1. import fs from "fs";
  2. import {
  3.   fileURLToPath
  4. } from "url";
  5. import {
  6.   join,
  7.   dirname
  8. } from "path";
  9.  
  10. const __dirname = dirname(fileURLToPath(
  11.   import.meta.url));
  12. const htmlPath = join(__dirname, "../html");
  13.  
  14. const resHandler = (req, res, next) => {
  15.   fs.readFile(`${htmlPath}/home.html`, "utf8", (err, data) => {
  16.     res.send(data);
  17.   });
  18. };
  19.  
  20. export default resHandler;
Promises with Node.js core APIs

Originally, core APIs for node.js used callbacks. With the introduction of promises in core APIs, we can now use then/catch blocks or async/await syntax to resolve our logic. For example, we can now use all functions present in the file system API by importing the fs/promises API. 

  1. import fs from "fs/promises";
  2. import {
  3.   fileURLToPath
  4. } from "url";
  5. import {
  6.   join,
  7.   dirname
  8. } from "path";
  9.  
  10. const __dirname = dirname(fileURLToPath(
  11.   import.meta.url));
  12. const htmlPath = join(__dirname, "../html");
  13.  
  14. const resHandler = async(req, res, next) => {
  15.   try {
  16.     const data = await fs.readFile(`${htmlPath}/home.html`, "utf8");
  17.     res.send(data);
  18.   } catch (err) {
  19.     console.log(err);
  20.   }
  21. };
  22.  
  23. export default resHandler;
Try it

To run the app, in your terminal run the following commands: 

  1. // rest API example
  2. cd assets/scripts/lesson-29
  3.  
  4. npm install // if you have not installed dependencies
  5.  
  6. cp .env.example .env // copy env variables into variable file and change their values
  7.  
  8. npm run serve // starts application
