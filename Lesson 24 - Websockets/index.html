Understanding Websockets & Socket.io

In this section, I learned about Websockets. What are Websockets? Wikipedia quotes it as "a communications protocol providing full duplex communication channels over a single TCP connection". They allow a long-held single TCP socket connection to be established between the client and server, enabling bi-directional, full duplex messages to be instantly distributed with little overhead, resulting in a highly efficient, very low latency connection.(Sourced from Pusher→)

Setting up websockets

To setup websockets we can use one of the most used libraries, socket.io→. The library is installed on both the backend(npm install --save socket.io) and frontend npm install --save socket.io-client. We begin by configuring the backend API to establish a TCP connection whenever the frontend connects to the backend. We initliaze this with the express server instance so as to listen for any connections. 

  1. // utils/socket.js
  2. let io;
  3.  
  4. module.exports = {
  5.   init(server) {
  6.     io = require('socket.io')(server, {
  7.       cors: {
  8.         origin: 'http://localhost:3000',
  9.         methods: ["GET", "POST"],
  10.         credentials: true
  11.       }
  12.     });
  13.  
  14.     return io;
  15.   },
  16.   getIO() {
  17.     if (!io) {
  18.       throw new Error('Socket.io not initialized!');
  19.     }
  20.  
  21.     return io;
  22.   }
  23. };
  24.  
  25. // app.js
  26. const express = require('express');
  27. const mongoose = require('mongoose');
  28.  
  29. const app = express();
  30. const port = 'APP_PORT';
  31. const mongodb_uri = 'MONGODB_URI';
  32.  
  33. mongoose.connect(
  34.   mongodb_uri, {
  35.     useNewUrlParser: true,
  36.     useUnifiedTopology: true,
  37.     useFindAndModify: false
  38.   }
  39. ).then(() => {
  40.   console.log('MONGO CONNECTED!');
  41.  
  42.   // mount the nodejs app to a port
  43.   const server = app.listen(port, () => {
  44.     console.log(`Server running at: ${APP_BASE_URL}`);
  45.     console.log('Hit CTRL-C to stop the server');
  46.   });
  47.  
  48.   const io = require('./utils/socket').init(server);
  49.  
  50.   io.on('connection', socket => {
  51.     console.log('Socket client connected');
  52.   });
  53. }).catch(err => console.log(err));
Connecting the frontend and backend

With the server instance setup, we can now add logic to send data to a defined channel of our choice after a HTTP request is performed. Such as when creating a post like below. This will then send the passed data to the channel which the frontend will pick up and use it to update the DOM. 

  1. const User = require('../models/user');
  2. const socket = require('../utils/socket');
  3. const {
  4.   validationResult
  5. } = require('express-validator');
  6.  
  7. exports.addPost = async(req, res, next) => {
  8.   const image = req.file;
  9.   const {
  10.     title,
  11.     content
  12.   } = req.body;
  13.   const errors = validationResult(req);
  14.  
  15.   try {
  16.     // validation logic...
  17.  
  18.     const post = new Post({
  19.       title,
  20.       content,
  21.       imageUrl,
  22.       creator: req.userId
  23.     });
  24.  
  25.     await post.save();
  26.  
  27.     const user = await User.findById(req.userId);
  28.  
  29.     user.posts.push(post);
  30.     await user.save();
  31.  
  32.     /**
  33.      * Creates an event channel with a name
  34.      * and object of our choice
  35.      */
  36.     socket.getIO().emit('posts', {
  37.       action: 'create',
  38.       post: {
  39.         ...post._doc,
  40.         creator: user
  41.       }
  42.     });
  43.  
  44.     res.status(201).json({
  45.       status: 'success',
  46.       data: post,
  47.       creator: user
  48.     });
  49.   } catch (err) {
  50.     if (!err.status_code) {
  51.       err.status_code = 500;
  52.     }
  53.  
  54.     next(err);
  55.   }
  56. };

The frontend receives a "hook" through the TCP connection where we can now perform our desired action. 

  1. socket.on('posts', data => {
  2.   if (data.action === 'create') {
  3.     this.addPost(data.post);
  4.   } else if (data.action === 'update') {
  5.     this.updatePost(data.post)
  6.   } else if (data.action === 'delete') {
  7.     this.loadPosts()
  8.   }
  9. });
